# Skip specific (version, target) builds. generate_all.sh skips just the listed
# targets for that version and still builds the rest, so one broken platform
# doesn't drop the whole release.
#
# Format: "<version> <goos/goarch> [<goos/goarch>...]" per line. "#" starts a
# comment; blank lines are ignored; the leading "go" on the version is optional.
#
# go1.23.0–go1.23.2: windows/amd64 fails to link with a nosplit stack overflow
#   (toolchain regression fixed in go1.23.3). linux/amd64, linux/arm64 and
#   darwin/arm64 build fine, so we only skip windows/amd64 for these.

1.23.1 windows/amd64
